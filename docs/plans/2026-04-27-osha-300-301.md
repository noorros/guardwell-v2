# OSHA 300/301 + sharpsDeviceType Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Restore OSHA recordkeeping completeness — generate **OSHA Form 300** (annual log of work-related injuries/illnesses, multi-row) and **OSHA Form 301** (single-incident detail report). Add the `sharpsDeviceType` field to support 29 CFR §1910.1030 Bloodborne Pathogens Standard sharps tracking. Wire `oshaDaysRestricted` into the report-incident form (currently hardcoded null). Smaller scope than the original audit estimate — most OSHA injury fields already exist on the schema.

**Architecture:** Same pattern as the chunk 2 breach memo PDF: `@react-pdf/renderer` document component in `src/lib/audit/`, GET route under `src/app/api/audit/`. The OSHA 300 is practice-wide (list-style report on `/audit/reports`); the OSHA 301 is per-incident (CTA on incident detail page like the breach memo). Schema change is additive (`sharpsDeviceType String?`); event payload extends `INCIDENT_REPORTED` v1 with optional `sharpsDeviceType` (no version bump).

**Tech Stack:** `@react-pdf/renderer` (already used by 8 existing PDF routes), Prisma 5.22, vi.mock("@/lib/auth") test pattern from chunk 2, existing event-sourcing pipeline.

**Pre-existing infrastructure already in place** (verified by survey, do not rebuild):
- All 5 primary OSHA fields on Incident: `oshaBodyPart`, `oshaInjuryNature`, `oshaOutcome`, `oshaDaysAway`, `oshaDaysRestricted` (yes — already in schema, even though the form doesn't yet collect days-restricted)
- The `IncidentReportForm` already has the `isOsha = type === "OSHA_RECORDABLE"` toggle + conditional OSHA subform with body-part, injury-nature, outcome enum, days-away inputs
- `INCIDENT_REPORTED` event payload already includes all 5 OSHA fields
- `INCIDENT_OSHA_RECORDABLE` derivation rule already flips `OSHA_300_LOG` requirement to COMPLIANT on incident report (verified in `tests/integration/osha-300-log.test.ts`)
- `OshaOutcomeEnum` exists in `actions.ts` (DEATH | DAYS_AWAY | RESTRICTED | OTHER_RECORDABLE | FIRST_AID)
- `Incident.reportedByUserId` → `User` join works for "employee name" placeholder on the OSHA 300 log

// src/app/api/audit/osha-300/route.tsx
//
// GET /api/audit/osha-300?year=YYYY
// Renders the OSHA Form 300 annual log PDF for a calendar year.
// Defaults to the current calendar year if `year` is not provided.

import { NextResponse } from "next/server";
import { renderToBuffer } from "@react-pdf/renderer";
import { requireUser } from "@/lib/auth";
import { getPracticeUser } from "@/lib/rbac";
import { db } from "@/lib/db";
import { Osha300Document, type Osha300Row } from "@/lib/audit/osha-300-pdf";

export const maxDuration = 120;

export async function GET(req: Request) {
  try {
    await requireUser();
  } catch {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }
  const pu = await getPracticeUser();
  if (!pu) {
    return NextResponse.json({ error: "Forbidden" }, { status: 403 });
  }

  const url = new URL(req.url);
  const yearParam = url.searchParams.get("year");
  const year = yearParam
    ? Number.parseInt(yearParam, 10)
    : new Date().getUTCFullYear();
  if (Number.isNaN(year) || year < 2000 || year > 2100) {
    return NextResponse.json(
      { error: "Invalid year parameter" },
      { status: 400 },
    );
  }

  const yearStart = new Date(`${year}-01-01T00:00:00Z`);
  const yearEnd = new Date(`${year + 1}-01-01T00:00:00Z`);

  const incidents = await db.incident.findMany({
    where: {
      practiceId: pu.practiceId,
      type: "OSHA_RECORDABLE",
      discoveredAt: { gte: yearStart, lt: yearEnd },
    },
    orderBy: { discoveredAt: "asc" },
    select: {
      id: true,
      discoveredAt: true,
      oshaInjuryNature: true,
      oshaOutcome: true,
      oshaDaysAway: true,
      oshaDaysRestricted: true,
      reportedByUserId: true,
    },
  });

  // Schema doesn't define a relation, just the scalar FK. Fetch all
  // unique reporters in one query to keep this O(1) regardless of the
  // number of OSHA-recordable cases.
  const reporterIds = Array.from(
    new Set(incidents.map((i) => i.reportedByUserId)),
  );
  const reporters = reporterIds.length
    ? await db.user.findMany({
        where: { id: { in: reporterIds } },
        select: { id: true, firstName: true, lastName: true, email: true },
      })
    : [];
  const reporterById = new Map(reporters.map((u) => [u.id, u]));

  const rows: Osha300Row[] = incidents.map((i, idx) => {
    const r = reporterById.get(i.reportedByUserId);
    const employeeName = r
      ? [r.firstName, r.lastName].filter(Boolean).join(" ") || r.email
      : null;
    return {
      caseNumber: String(idx + 1).padStart(3, "0"),
      injuryDate: i.discoveredAt,
      employeeName,
      injuryNature: i.oshaInjuryNature,
      outcome: i.oshaOutcome,
      daysAway: i.oshaDaysAway,
      daysRestricted: i.oshaDaysRestricted,
    };
  });

  const pdfBuffer = await renderToBuffer(
    <Osha300Document
      input={{
        practiceName: pu.practice.name,
        practiceState: pu.practice.primaryState,
        year,
        generatedAt: new Date(),
        rows,
      }}
    />,
  );

  return new NextResponse(new Uint8Array(pdfBuffer), {
    status: 200,
    headers: {
      "Content-Type": "application/pdf",
      "Content-Disposition": `inline; filename="osha-300-${year}.pdf"`,
      "Cache-Control": "private, no-store",
    },
  });
}

// Helper to extract human-readable text from a react-pdf-generated buffer.
// React-PDF compresses content streams with FlateDecode and emits text
// inside hex-encoded `<…>` TJ operators. This helper inflates streams +
// decodes hex literals so test assertions can match substrings of the
// rendered body content (not just metadata).

import { inflateSync } from "node:zlib";

export function extractInflatedText(buf: Uint8Array): string {
  const raw = Buffer.from(buf).toString("latin1");
  const parts: string[] = [];

  const streamRegex = /\/Filter\s*\/FlateDecode[^]*?stream\r?\n([^]*?)\r?\nendstream/g;
  let m: RegExpExecArray | null;
  while ((m = streamRegex.exec(raw)) !== null) {
    try {
      const compressed = Buffer.from(m[1], "latin1");
      const inflated = inflateSync(compressed).toString("latin1");
      parts.push(inflated);
    } catch {
      // skip streams that fail to inflate (e.g. font subsets)
    }
  }

  const combined = parts.join("\n");
  // Decode hex literals: `<48495020414120536f6d65>` → "HIP AA Some"
  return combined.replace(/<([0-9a-fA-F]+)>/g, (_full, hex: string) => {
    let out = "";
    for (let i = 0; i < hex.length; i += 2) {
      const charCode = Number.parseInt(hex.slice(i, i + 2), 16);
      if (Number.isFinite(charCode) && charCode > 0) {
        out += String.fromCharCode(charCode);
      }
    }
    return out;
  });
}
